# /etc/network/interfaces

# Default source for additional configs
# source /etc/network/interfaces.d/*

auto eth0
iface eth0 inet static
    address 192.168.5.100
    netmask 255.255.255.0
    gateway 192.168.5.1

auto eth1
iface eth1 inet static
    address 192.168.1.100
    netmask 255.255.255.0

    # --- anycast + policy routing setup ---
    up ip route add 192.168.1.0/24 dev eth1 scope link table anycast
    up ip route add 192.168.1.3 dev eth1 scope link table anycast
    up ip route add default via 192.168.1.3 dev eth1 table anycast
    up ip rule add from 192.168.1.100/32 lookup anycast

    # --- global default still via eth0 ---
    up ip route replace default via 192.168.5.1 dev eth0
